annex agree {**
  eq numberServicesTrue : int = 
    (if inpGCASM then 1 else 0) + 
    (if inpSAM then 1 else 0) + 
    (if inpACASM then 1 else 0) + 
    (if inpGFM then 1 else 0) + 
    (if inpWxM then 1 else 0) + 
    (if inpFLS then 1 else 0);
  
  guarantee "number of active services is between 0 and 6": 
    numberServicesTrue >= 0 and numberServicesTrue <= 6;
  
  eq dynamicDecision : bool = 
    if inpGCASImpact > inpACASImpact and 
       (inpGCASRecovery + inpACASRecovery) >= 0.0 
    then true else false;
  
  guarantee "dynamic decision is ACAS when conditions are met": 
    dynamicDecision => outDec = 1;
  
  guarantee "dynamic decision is GCAS when conditions are not met": 
    not dynamicDecision => outDec = 0;
  
  guarantee "output is dynamic decision when at least 2 services active": 
    numberServicesTrue >= 2 => outDec = (if dynamicDecision then 1 else 0);
  
  guarantee "output is static priority when less than 2 services active": 
    numberServicesTrue < 2 => outDec = 0;
**};
