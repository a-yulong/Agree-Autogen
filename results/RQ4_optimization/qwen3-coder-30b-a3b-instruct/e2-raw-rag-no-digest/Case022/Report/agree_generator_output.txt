annex agree {**
  eq heating_request : bool = user_request.heater_on and not user_request.cooler_on;
  eq cooling_request : bool = user_request.cooler_on and not user_request.heater_on;
  guarantee "heater_on enabled only if heating requested and tempin <= 20": heater_on => (heating_request and tempin <= 20);
  guarantee "cooler_on enabled only if cooling requested and tempin > 0": cooler_on => (cooling_request and tempin > 0);
**};
