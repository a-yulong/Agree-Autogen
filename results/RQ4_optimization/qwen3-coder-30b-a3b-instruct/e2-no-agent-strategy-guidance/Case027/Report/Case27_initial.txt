package gcas_single
public
	with base_types;
	with gcas_nodes;
	
	system GCAS_singleton
		features
			altitude: in data port real;
			timeToRecovery: out data port real;
	annex agree {**
	  const minHeightAllowed = 200.0;
	  var t: real = 0.0;
	  eq t = 0.0 -> pre(t) + 0.1;
  
	  guarantee "R6": timeToRecovery = 3.7;
  
	  guarantee "R1": (t >= 3.7) => (altitude > minHeightAllowed);
  
	  guarantee "R2": (t >= 3.7) => (altitude > pre(altitude));
	**};
	end GCAS_singleton;
	
	system implementation GCAS_singleton.impl
	end GCAS_singleton.impl;
	
	
	-- Type Aliases
	
	data int extends base_types::Integer
	end int;
	
	data real extends base_types::Float
	end real;
	
	data bool extends base_types::Boolean
	end bool;
	
end gcas_single;
