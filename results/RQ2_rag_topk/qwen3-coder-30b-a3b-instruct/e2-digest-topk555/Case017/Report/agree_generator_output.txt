annex agree {**
  const threshold : real = 1.0;
  
  eq numberServicesTrue : int = 
    (if inpGCASM then 1 else 0) +
    (if inpSAM then 1 else 0) +
    (if inpACASM then 1 else 0) +
    (if inpGFM then 1 else 0) +
    (if inpWxM then 1 else 0) +
    (if inpFLS then 1 else 0);
  
  guarantee "dynamic_decision_ACAS": 
    (inpGCASImpact > inpACASImpact and (inpGCASRecovery + inpACASRecovery) >= threshold) => outDec = 1;
  
  guarantee "dynamic_decision_GCAS": 
    (inpGCASImpact <= inpACASImpact or (inpGCASRecovery + inpACASRecovery) < threshold) => outDec = 0;
  
  guarantee "priority_based_decision": 
    numberServicesTrue < 2 => outDec = 0;
**};
