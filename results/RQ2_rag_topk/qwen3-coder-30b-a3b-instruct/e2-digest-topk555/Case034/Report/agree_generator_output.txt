annex agree {**
  assume "threatDetected is Boolean": threatDetected = true or threatDetected = false;
  assume "threatAltitude is Float": threatAltitude >= 0.0;
  assume "ownAltitude is Float": ownAltitude >= 0.0;
  assume "ownLongitude is Int": ownLongitude >= 0;
  assume "threatLongitude is Int": threatLongitude >= 0;
  guarantee "priority based on longitude when altitudes equal and threat detected": 
    threatDetected and ownAltitude = threatAltitude => 
      (ownLongitude > threatLongitude => requiredVerticalVelocity = 50) and 
      (ownLongitude <= threatLongitude => requiredVerticalVelocity = -50);
  guarantee "top intrusion avoidance when own altitude lower than threat": 
    threatDetected and ownAltitude < threatAltitude => requiredVerticalVelocity = -10;
  guarantee "bottom intrusion avoidance when own altitude higher than threat": 
    threatDetected and ownAltitude > threatAltitude => requiredVerticalVelocity = 10;
**};
