package gcas_single
public
	with base_types;
	with gcas_nodes;
	
	system GCAS_singleton
		features
			altitude: in data port real;
			timeToRecovery: out data port real;
	annex agree {**
	  assume "initial_t_value": t = 0.0;
	  assume "t_step_size": t' = t + 0.1;
	  const "minHeightAllowed": real = 200.0;
	  guarantee "timeToRecovery_constant": timeToRecovery = 3.7;
	  guarantee "height_above_min_when_t_ge_3_7": 
	    (t >= 3.7) => (height > minHeightAllowed);
	  guarantee "pull_up_trend_when_t_ge_3_7": 
	    (t >= 3.7) => (height > pre(height));
	**};
	end GCAS_singleton;
	
	system implementation GCAS_singleton.impl
	end GCAS_singleton.impl;
	
	
	-- Type Aliases
	
	data int extends base_types::Integer
	end int;
	
	data real extends base_types::Float
	end real;
	
	data bool extends base_types::Boolean
	end bool;
	
end gcas_single;
