package geofence_single
public
	
	with geofence_nodes;
	with base_types;
	
	system GeoFence_singleton
		features
			timeToRecovery: out data port real;
	annex agree {**
	  eq t : real = 0.0;
	  const step_size : real = 0.1;
	  const recovery_time : real = 1.7;
  
	  assume "initial_t": t = 0.0;
	  assume "step_size": step_size = 0.1;
	  assume "recovery_time": recovery_time = 1.7;
  
	  guarantee "time_to_recovery": timeToRecovery = 1.7;
  
	  guarantee "recovery_condition": 
	    (t >= recovery_time) => (timeToRecovery > 1.7);
	**};
	end GeoFence_singleton;
	
	system implementation GeoFence_singleton.impl
	end GeoFence_singleton.impl;
	
	-- Type Aliases
	
	data int extends base_types::Integer
	end int;
	
	data real extends base_types::Float
	end real;
	
end geofence_single;
