package geofence_single
public
	
	with geofence_nodes;
	with base_types;
	
	system GeoFence_singleton
		features
			timeToRecovery: out data port real;
	annex agree {**
	  assume "initial_t_value": t = 0.0;
	  assume "step_size": delta_t = 0.1;
	  guarantee "recovery_condition": (t >= recovery_time) => (distanceFromFence > distanceFromFence_previous);
	  guarantee "output_assignment": timeToRecovery = 1.7;
	**};
	end GeoFence_singleton;
	
	system implementation GeoFence_singleton.impl
	end GeoFence_singleton.impl;
	
	-- Type Aliases
	
	data int extends base_types::Integer
	end int;
	
	data real extends base_types::Float
	end real;
	
end geofence_single;
