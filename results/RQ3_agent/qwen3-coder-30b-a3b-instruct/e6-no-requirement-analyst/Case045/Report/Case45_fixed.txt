package Integer_Toy
public 
   with Base_Types;
   
system A
	features
		Input: in data port Base_Types::Integer;
		Output: out data port Base_Types::Integer;
end A ; 

system B
	features
		Input: in data port Base_Types::Integer;
		Output: out data port Base_Types::Integer;
end B ; 

system C
	features
		Input1: in data port Base_Types::Integer;
		Input2: in data port Base_Types::Integer;
		Output: out data port Base_Types::Integer;
annex agree {**
  const initial_mode_geq_0 : bool = true;
  const mode_increasing : bool = true;
  guarantee "initial_mode": initial_mode_geq_0;
  guarantee "monotonic_mode": mode_increasing;
  guarantee "output_logic": (Output = Input1 + Input2) and (Output = 0);
**};
end C ; 
   
system top_level
	features
		Input: in data port Base_Types::Integer;
		Output: out data port Base_Types::Integer;
end top_level;

system implementation top_level.Impl
	subcomponents
		A_sub : system A ; 
		B_sub : system B ; 
		C_sub : system C ; 
	connections
		IN_TO_A : port Input -> A_sub.Input;
		A_TO_B : port A_sub.Output -> B_sub.Input;
		A_TO_C : port A_sub.Output -> C_sub.Input1;
		B_TO_C : port B_sub.Output -> C_sub.Input2;
		C_TO_Output : port C_sub.Output -> Output; 
		
end top_level.Impl;
	
end Integer_Toy;
