package gcas_single
public
	with base_types;
	with gcas_nodes;
	
	system GCAS_singleton
		features
			altitude: in data port real;
			timeToRecovery: out data port real;
	annex agree {**
			  assume "initial_t": timeToRecovery = 0.0;
			  assume "step_size": altitude = 0.1;
			  const minHeightAllowed : real = 200.0;
			  const recovery_time : real = 3.7;
			  guarantee "height_bound": (timeToRecovery >= recovery_time) => (altitude > minHeightAllowed);
			  guarantee "pull_up_trend": (timeToRecovery >= recovery_time) => (altitude > minHeightAllowed);
			**};
	end GCAS_singleton;
	
	system implementation GCAS_singleton.impl
	end GCAS_singleton.impl;
	
	
	-- Type Aliases
	
	data int extends base_types::Integer
	end int;
	
	data real extends base_types::Float
	end real;
	
	data bool extends base_types::Boolean
	end bool;
	
end gcas_single;
