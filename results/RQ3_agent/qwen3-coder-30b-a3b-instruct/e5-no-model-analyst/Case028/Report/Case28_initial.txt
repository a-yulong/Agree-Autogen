package gcas_single
public
	with base_types;
	with gcas_nodes;
	
	system GCAS_singleton
		features
			altitude: in data port real;
			timeToRecovery: out data port real;
	annex agree {**
	  const "minHeightAllowed": real = 200.0;
	  const "recoveryTime": real = 3.7;
	  eq "t_next": real = 0.0 -> pre(t) + 0.1;
	  guarantee "output_timeToRecovery": timeToRecovery = 3.7;
	  guarantee "height_greater_than_minHeightAllowed": 
	    (t >= recoveryTime) => (height > minHeightAllowed);
	  guarantee "pull_up_trend": 
	    (t >= recoveryTime) => (height > pre(height));
	**};
	end GCAS_singleton;
	
	system implementation GCAS_singleton.impl
	end GCAS_singleton.impl;
	
	
	-- Type Aliases
	
	data int extends base_types::Integer
	end int;
	
	data real extends base_types::Float
	end real;
	
	data bool extends base_types::Boolean
	end bool;
	
end gcas_single;
