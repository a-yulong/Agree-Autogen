annex agree {**
  assume "service_inputs_valid": 
    inpGCASM in {true, false} and 
    inpSAM in {true, false} and 
    inpACASM in {true, false} and 
    inpGFM in {true, false} and 
    inpWxM in {true, false} and 
    inpFLS in {true, false};
  
  guarantee "number_services_true": 
    numberServicesTrue = 
      (if inpGCASM then 1 else 0) +
      (if inpSAM then 1 else 0) +
      (if inpACASM then 1 else 0) +
      (if inpGFM then 1 else 0) +
      (if inpWxM then 1 else 0) +
      (if inpFLS then 1 else 0);
  
  guarantee "dynamic_decision_logic": 
    dynamicDecide = 
      (if (inpGCASImpact > inpACASImpact) and 
           (inpGCASRecovery + inpACASRecovery >= 0.0) 
       then ACAS 
       else GCAS);
  
  guarantee "output_decision": 
    (numberServicesTrue >= 2) => (outDec = dynamicDecide) and
    (numberServicesTrue < 2) => (outDec = static_priority_value);
  
  const "static_priority_value" = 0;
  
  eq "dynamicDecide" = ACAS or dynamicDecide = GCAS;
  
  eq "ACAS" = 1;
  
  eq "GCAS" = 2;
**};
